typedef unsigned int uint32;

uint32 Crc32Table[256];

int MakeTable()
{
	uint32 c;
	
	for(int i = 0; i < 256; ++i)
	{
		c = static_cast<uint32>(i);
		
		for(int bit = 0; bit < 8; ++bit)
		{
			if(c & 1)
			{
				c = (c >> 1) ^ 0xEDB88320;
			}
			else
			{
				c = c >> 1;
			}
			Crc32Table[i] = c;
		}
	}
}

uint32 CRC32(const unsigned char * buf, unsigned int size)
{
	unsigned int i, crc = 0xFFFFFFFF;
	
	for(i = 0; i < size; ++i)
	{
		crc = Crc32Table[(crc ^ buf[i]) & 0xFF] ^ (crc >> 8);
	}
	return crc ^ 0xFFFFFFFF;
}
